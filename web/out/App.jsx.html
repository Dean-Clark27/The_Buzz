import React, { useState } from 'react'
import './App.css'
import MessageContainer from './components/MessageContainer'
import Navbar from './components/Navbar'
import AddMessageForm from './components/AddMessageForm'

/**
 * Main application component that handles state and renders 
 * the navigation bar, message form, and message container.
 *
 * @component
 */
function App() {
  /**
   * Holds the message data.
   * @type {Object}
   */
  const [messageData, setMessageData] = useState({});

  /**
   * Controls the visibility of the add message form.
   * @type {boolean}
   */
  const [isAddVisible, setIsAddVisible] = React.useState(false);

  /**
   * Toggles the visibility of the AddMessageForm component.
   */
  const toggleAddVisibility = () => {
    setIsAddVisible(!isAddVisible);
  };

  return (
    <>
      {/* Navbar component with a toggle function for form visibility */}
      <Navbar toggleForm={toggleAddVisibility}/>
      {/* AddMessageForm component, conditionally visible */}
      <AddMessageForm isVisible={isAddVisible}/>
      {/* MessageContainer to display messages */}
      <MessageContainer />
    </>
  )
}

export default App
